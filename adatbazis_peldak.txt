--példa a raktárhoz

SELECT
    t.termek_id,
    t.csapatnev,
    t.leiras,
    m.meret_nev,
    r.mennyiseg
FROM raktar r
JOIN termekek t ON r.termek_id = t.termek_id
JOIN meretek m ON r.meret_id = m.meret_id
ORDER BY t.termek_id, m.meret_id
LIMIT 30;

-- példa a meccsekhez

SELECT
    m.meccs_id,
    c1.nev AS hazai_csapat,
    c2.nev AS vendeg_csapat,
    s.nev AS stadion,
    m.datum,
    m.kezdesi_ido,
    m.jegyar
FROM meccsek m
JOIN csapatok c1 ON m.hazai_csapat_id = c1.csapat_id
JOIN csapatok c2 ON m.vendeg_csapat_id = c2.csapat_id
JOIN stadionok s ON m.stadion_id = s.stadion_id
ORDER BY m.datum, m.kezdesi_ido;

-- példa az utazásokhoz

SELECT
    u.utazas_id,
    c1.nev AS hazai_csapat,
    c2.nev AS vendeg_csapat,
    u.indulas_varos,
    u.erkezes_varos,
    u.indulas_idopont,
    u.erkezes_idopont,
    u.ar
FROM utazasok u
JOIN meccsek m ON u.meccs_id = m.meccs_id
JOIN csapatok c1 ON m.hazai_csapat_id = c1.csapat_id
JOIN csapatok c2 ON m.vendeg_csapat_id = c2.csapat_id
ORDER BY u.utazas_id;

--példa a szállásokhoz

SELECT COUNT(*) AS szallasok_szama FROM szallasok;
SELECT * FROM szallasok ORDER BY varos, nev;

--példa a foglaláshoz

SELECT
    f.foglalas_id,
    u.nev AS felhasznalo,
    c1.nev AS hazai_csapat,
    c2.nev AS vendeg_csapat,
    sz.nev AS szallas_nev,
    f.szallas_kezdo_datuma,
    f.szallas_vege_datuma,
    f.darabszam,
    f.fizetve
FROM foglalasok f
JOIN felhasznalok u ON f.felhasznalo_id = u.felhasznalo_id
JOIN utazasok ut ON f.utazas_id = ut.utazas_id
JOIN meccsek m ON ut.meccs_id = m.meccs_id
JOIN csapatok c1 ON m.hazai_csapat_id = c1.csapat_id
JOIN csapatok c2 ON m.vendeg_csapat_id = c2.csapat_id
JOIN szallasok sz ON f.szallas_id = sz.szallas_id
ORDER BY f.foglalas_id;


--példa rendelések

SELECT
    r.rendeles_id,
    r.nev AS vasarlo_nev,
    r.email,
    r.datum,
    r.allapot,
    r.osszeg,
    r.szallitas_mod,
    r.fizetes_mod
FROM rendelesek r
ORDER BY r.datum DESC;

--példa tételek
SELECT
    r.rendeles_id,
    r.nev AS vasarlo_nev,
    t.csapatnev,
    t.leiras,
    m.meret_nev AS meret,
    rt.darabszam,
    rt.ar
FROM rendeles_tetelek rt
JOIN rendelesek r ON rt.rendeles_id = r.rendeles_id
JOIN termekek t ON rt.termek_id = t.termek_id
JOIN meretek m ON rt.meret_id = m.meret_id
ORDER BY r.rendeles_id, t.csapatnev, t.leiras;

--példa fizetések

SELECT COUNT(*) AS fizetesek_szama FROM fizetes;

--összekötött példa az előző 3-hoz:
SELECT
    r.rendeles_id,
    r.nev AS vevo_nev,
    t.csapatnev,
    t.leiras,
    m.meret_nev,
    rt.darabszam,
    rt.ar,
    f.fizetes_allapot
FROM rendelesek r
JOIN rendeles_tetelek rt ON r.rendeles_id = rt.rendeles_id
JOIN termekek t ON rt.termek_id = t.termek_id
JOIN meretek m ON rt.meret_id = m.meret_id
LEFT JOIN fizetes f ON r.rendeles_id = f.rendeles_id
ORDER BY r.rendeles_id, rt.rendeles_tetel_id;


--példa a raktárhoz (full raktár)
SELECT
    r.raktar_id,
    t.csapatnev,
    t.leiras,
    m.meret_nev,
    r.mennyiseg,
    t.alap_ar_ft,
    t.kep_url
FROM raktar r
JOIN termekek t ON r.termek_id = t.termek_id
JOIN meretek m ON r.meret_id = m.meret_id
ORDER BY t.csapatnev, t.leiras, m.meret_nev;

--raktár példa egy adott csapathoz
SELECT
    t.csapatnev,
    t.leiras,
    m.meret_nev AS meret,
    r.mennyiseg
FROM raktar r
JOIN termekek t ON r.termek_id = t.termek_id
JOIN meretek m ON r.meret_id = m.meret_id
WHERE t.csapatnev = 'Arsenal FC'
ORDER BY t.leiras, m.meret_nev;

--termékek példa
SELECT
    t.termek_id,
    t.csapatnev,
    t.leiras,
    t.alap_ar_ft,
    SUM(r.mennyiseg) AS osszes_keszlet
FROM termekek t
LEFT JOIN raktar r ON t.termek_id = r.termek_id
GROUP BY t.termek_id, t.csapatnev, t.leiras, t.alap_ar_ft
ORDER BY t.csapatnev, t.leiras;

--utazás + foglalás komplett példa
SELECT
    f.foglalas_id,
    u.nev AS felhasznalo,
    c1.nev AS hazai_csapat,
    c2.nev AS vendeg_csapat,
    s.nev AS stadion,
    ut.indulas_varos,
    ut.erkezes_varos,
    sz.nev AS szallas_nev,
    f.szallas_kezdo_datuma,
    f.szallas_vege_datuma,
    f.darabszam,
    f.fizetve
FROM foglalasok f
JOIN felhasznalok u ON f.felhasznalo_id = u.felhasznalo_id
JOIN utazasok ut ON f.utazas_id = ut.utazas_id
JOIN meccsek m ON ut.meccs_id = m.meccs_id
JOIN csapatok c1 ON m.hazai_csapat_id = c1.csapat_id
JOIN csapatok c2 ON m.vendeg_csapat_id = c2.csapat_id
JOIN stadionok s ON m.stadion_id = s.stadion_id
JOIN szallasok sz ON f.szallas_id = sz.szallas_id
ORDER BY f.foglalas_id;
